To download the libraries for ubuntu use the following commands

    sudo apt-get update

opencv library
    
    sudo apt-get install libopencv-dev

tesseract

    sudo apt-get install -y \
    tesseract-ocr \
    libtesseract-dev \
    tesseract-ocr-eng

libleptonica

    sudo apt-get install -y libleptonica-dev
